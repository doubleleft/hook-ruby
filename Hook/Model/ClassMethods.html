def field name, options = {}
  define_attribute_method name

  # Define getter
  define_method name do
    # self.instance_variable_get("@#{name}")
    attributes[name]
  end

  # Define setter
  define_method "#{name}=" do |value|
    self.send(:#{name}_will_change!")
    attributes[name] = value
  end
end
