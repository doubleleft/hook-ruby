def initialize options = {}
  @app_id = options.delete(:app_id)
  @key = options.delete(:key)
  @endpoint = options.delete(:endpoint) || options.delete(:url)
  @endpoint = "#{@endpoint}/" unless @endpoint.end_with? "/"

  @logger = options.delete(:logger)
end
